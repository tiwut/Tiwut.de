#include <iostream> // 'Input/Output Stream' library for console I/O

// The main function is the entry point of every C++ program.
int main() {
    // 'std::cout' is the standard character output stream.
    // The '<<' operator is the 'stream insertion operator'.
    std::cout << "Hello, C++!" << std::endl;

    // 'return 0' indicates successful execution to the operating system.
    return 0;
}

    MyCode::printMessage(); // Calling function from our namespace
    return 0;
}

std::cout << *pNumber; // Dereference the pointer to get the value (42)

delete pNumber; // CRITICAL: Deallocate the memory to prevent a leak
pNumber = nullptr; // Good practice: Set pointer to null after deletion


#include <memory>

{ // Some scope
    // Use std::make_unique (C++14) to create it.
    std::unique_ptr<int> pUnique = std::make_unique<int>(100);
    
    std::cout << *pUnique; // Use it like a raw pointer

} // pUnique goes out of scope here, memory is automatically freed. No delete needed!


#include <memory>

{
    std::shared_ptr<int> pShared1;
    {
        // Use std::make_shared (C++11)
        std::shared_ptr<int> pShared2 = std::make_shared<int>(200);
        pShared1 = pShared2; // Copying is allowed, ref count is now 2
    } // pShared2 goes out of scope, ref count is 1. Memory is NOT freed.
} // pShared1 goes out of scope, ref count becomes 0. Memory is freed NOW.
